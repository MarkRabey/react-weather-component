import {
  ArgsTable,
  Meta,
  Source,
  Story,
  Description,
} from '@storybook/addon-docs';
import {themeNames} from '../../../themes';
import defaultCustomTheme from './customTheme';
import customThemeControls, {argsToTheme} from './customThemeControls';

import Weather from '..';
let customTheme = defaultCustomTheme;

<Meta
  title="Styling/Custom Themes"
  component={Weather}
  argTypes={{
    apiKey: {table: {disable: true}},
    theme: {table: {disable: true}},
    ...customThemeControls,
  }}
/>

export const Template = args => {
  customTheme = argsToTheme(args);
  return (
    <Weather apiKey={process.env.STORYBOOK_API_KEY || ''} theme={customTheme} />
  );
};
